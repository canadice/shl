	teamID, 
	fhmID, 
	franchiseID, 
	leagueID, 
	name, 
	abbr, 
	primaryCol, 
	secondaryCol, 
	alt1Col, 
	alt2Col,

);

	playerID, 
	currentTeam, 
	user, 
	name, 
	nickname, 
	fhmName, 
	draftClass, 
	position, 
	activeStatus, 
	currentTPE, 
	shoots, 
	jerseyNr, 
	recruitedBy, 
	playerRender, 
	birthplace, 
	retired,

	FOREIGN KEY (currentTeam) REFERENCES teamData(teamID)
);
